4. Add `_onControllerChanged()`:
   ```dart
   void _onControllerChanged() {
     if (widget.controller.isStreaming) {
       // onUserMessageSent just fired — schedule anchor jump
       _anchorIndex = widget.itemCount - 1;
       SchedulerBinding.instance.addPostFrameCallback((_) {
         _executeAnchorJump();
       });
     } else {
       // onResponseComplete — stop filler updates
       _anchorActive = false;
     }
   }
   ```
5. Add `_executeAnchorJump()`:
   ```dart
   void _executeAnchorJump() {
     if (!_scrollController.hasClients) return;
     final pos = _scrollController.position;
     final box = _anchorKey.currentContext?.findRenderObject() as RenderBox?;
     final sentMsgHeight = box?.size.height ?? 60.0;

     final target = (pos.maxScrollExtent - pos.viewportDimension + sentMsgHeight)
         .clamp(pos.minScrollExtent, pos.maxScrollExtent);
     final initialFiller = math.max(0.0, pos.viewportDimension - sentMsgHeight);

     _fillerHeight.value = initialFiller;
     _scrollController.jumpTo(target);
     _anchorActive = true;
     _lastMaxScrollExtent = pos.maxScrollExtent;
   }
   ```
   IMPORTANT: Set `_anchorActive = true` AFTER `jumpTo` and filler assignment (per Pitfall 5 in research) so the scroll listener does not fire spuriously during the jump.

6. Add `_onScrollChanged()` with frame-throttled filler recomputation:
   ```dart
   void _onScrollChanged() {
     if (!_anchorActive || !_scrollController.hasClients) return;
     if (_fillerUpdateScheduled) return;
     _fillerUpdateScheduled = true;
     SchedulerBinding.instance.addPostFrameCallback((_) {
       _fillerUpdateScheduled = false;
       _recomputeFiller();
     });
   }
   ```
7. Add `_recomputeFiller()`:
   ```dart
   void _recomputeFiller() {
     if (!_anchorActive || !_scrollController.hasClients) return;
     final pos = _scrollController.position;
     final growth = pos.maxScrollExtent - _lastMaxScrollExtent;
     if (growth > 0) {
       _fillerHeight.value = math.max(0.0, _fillerHeight.value - growth);
       _lastMaxScrollExtent = pos.maxScrollExtent;
     }
   }
   ```
8. Wrap the `CustomScrollView` in a `NotificationListener<ScrollMetricsNotification>` (per Open Question 1 in research — extent changes without pixel changes need this):

   NotificationListener<ScrollMetricsNotification>(
     onNotification: (notification) {
       _onScrollChanged();  // reuse same throttled handler
       return false;
     },
     child: CustomScrollView(...)

   ```

     }
     return child;
   }
   ```
